\makeatletter
\newlength{\bi@label@len}
\newlength{\bi@left@move}
\newlength{\bilabelsep}
\newlength{\bicolumnsep}
\setlength{\bilabelsep}{2pt plus 0.5pt minus 0.5pt}
\setlength{\bicolumnsep}{6pt}
\newcounter{bienumii}
\renewcommand{\thebienumii}{\arabic{bienumii}}
\newcommand\labelbienumii{(\thebienumii)}
\newcommand*\bi@item@xx[2]{%
	\refstepcounter{bienumii}%
	\settowidth{\bi@label@len}{\labelbienumii}%
	\labelbienumii\hspace*{\bilabelsep}%
	\parbox[t]{\dimexpr \linewidth/2-\bi@label@len-\bilabelsep-\bicolumnsep/2-\bi@left@move\relax}{#1}\hspace*{6pt}%
	\refstepcounter{bienumii}%
	\settowidth{\bi@label@len}{\labelbienumii}%
	\labelbienumii\hspace*{\bilabelsep}%
	\parbox[t]{\dimexpr \linewidth/2-\bi@label@len-\bilabelsep-\bicolumnsep/2+\bi@left@move\relax}{#2}\par%
}
\newcommand*\bi@item@x[1]{%
	\refstepcounter{bienumii}%
	\settowidth{\bi@label@len}{\labelbienumii}%
	\labelbienumii\hspace*{\bilabelsep}%
	\parbox[t]{\dimexpr \linewidth-\bi@label@len-\bilabelsep\relax}{#1}\par%
}
\providecommand\biitemxx{\@latex@error{\noexpand\biitemxx used in wrong place.}{You dummy! How many times should I tell you!}}
\providecommand\biitemx{\@latex@error{\noexpand\biitemx used in wrong place.}{You dummy! How many times should I tell you!}}
% bienum's optional argument is the left-move length of the second column
\newenvironment{bienum}[1][0pt]{\par \linespread{1.6}\selectfont%
	\setcounter{bienumii}{0}%
	\setlength{\bi@left@move}{#1}%
	\let\biitemxx\bi@item@xx%
	\let\biitemx\bi@item@x%
}{}
\makeatother
